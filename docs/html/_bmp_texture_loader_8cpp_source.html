19 TextureData BmpTextureLoader::load(const std::string& filename) {
20  std::ifstream file;
21  file.open(filename.c_str(), std::ios_base::in | std::ios_base::binary);
22  if (!file.is_open()) {
23  Console::error() << "AnnaGraphics warning: unable to open texture file "
24  "\"" << filename << "\".";
25  return TextureData();
26  }
27 
28  file.exceptions(std::ifstream::eofbit | std::ifstream::failbit | std::
29  ifstream::badbit);
30  try {
31  size_t mark = readWord(file);
32  if (mark != 0x4d42) {
33  Console::error() << "AnnaGraphics warning: texture file \"" <<
34  filename << "\" has not BMP format.";
35  return TextureData();
36  }
37 
38  file.seekg(4 + 2 + 2, std::ios_base::cur);
39  size_t data_offset = readDoubleWord(file);
40 
41  file.seekg(4, std::ios_base::cur);
42  /* По описанию структуры BITMAPINFOHEADER и её расширений поля
43  * biWidth и biHeight имеют знаковый тип LONG, причём объясняется
44  * назначение этого лишь для поля biHeight.
45  * В данном загрузчике не используется информация, передаваемая знаками
46  * размеров, поэтому учитывается лишь их абсолютное значение.
47  */
48  size_t width = std::abs(readLong(file));
49  size_t height = std::abs(readLong(file));
50 
51  file.seekg(2, std::ios_base::cur);
52  size_t bits = readWord(file);
53  if (bits != 24 && bits != 32) {
54  Console::error() << "AnnaGraphics warning: texture file \"" <<
55  filename << "\" has not supported value of bits on pixel.";
56  return TextureData();
57  }
58  bool transparent = bits == 32;
59 
60  if (readDoubleWord(file) != 0) {
61  Console::error() << "AnnaGraphics warning: texture file \"" <<
62  filename << "\" has not supported compression.";
63  return TextureData();
64  }
65 
66  /* По описанию структуры BITMAPINFOHEADER и её расширений поле
67  * biSizeImage имеет беззнаковый тип DWORD, но на практике была
68  * встречена текстура с отрицательным значением данного поля. В
69  * следствие этого алгоритм чтения и проверки этого значения отличается
70  * от ожидаемого.
71  */
72  long data_size = readLong(file);
73  if (data_size <= 0) {
74  data_size = (width * (!transparent ? 3 : 4) + width % 4 ) * height;
75  }
76 
77  file.seekg(data_offset, std::ios_base::beg);
78  char* data = new char[data_size];
79  for (size_t i = 0; i < static_cast<size_t>(data_size); !transparent ? i
80  += 3 : i += 4)
81  {
82  data[i + 2] = readByte(file);
83  data[i + 1] = readByte(file);
84  data[i] = readByte(file);
85  if (transparent) {
86  data[i + 3] = readByte(file);
87  }
88  }
89 
90  return TextureData(data, Vector2D<size_t>(width, height), transparent);
91  } catch (const std::ifstream::failure& exception) {
92  (void)exception;
93  Console::error() << "AnnaGraphics warning: unable to read texture file "
94  "\"" << filename << "\".";
95  return TextureData();
96  } catch(const std::bad_alloc& exception) {
97  (void)exception;
98  Console::error() << "AnnaGraphics warning: in process of loading "
99  "texture file \"" << filename << "\" was requested too much memory;"
100  " perhaps some parameters of image have been read wrong.";
101  return TextureData();
102  }
103 }
