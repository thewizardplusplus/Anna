287 void OpenGlWindow::update(void) {
288  #ifdef OS_LINUX
289  XEvent event;
290 
291  while (XPending(display) > 0) {
292  XNextEvent(display, &event);
293  switch (event.type) {
294  case ConfigureNotify:
295  if (event.xconfigure.width != static_cast<int>(size.x) || event.
296  xconfigure.height != static_cast<int>(size.y))
297  {
298  resize(event.xconfigure.width, event.xconfigure.height);
299  }
300 
301  break;
302  case KeyPress: {
303  KeyCode::Types key = convertKeyCode(XLookupKeysym(&event.xkey,
304  0));
305  keys.push_back(key);
306 
307  break;
308  }
309  case KeyRelease: {
310  KeyCode::Types key = convertKeyCode(XLookupKeysym(&event.xkey,
311  0));
312  keys.remove(key);
313 
314  break;
315  }
316  case ButtonPress: {
317  ButtonCode::Types button = convertButtonCode(event.xbutton.
318  button);
319  buttons.push_back(button);
320 
321  break;
322  }
323  case ButtonRelease: {
324  ButtonCode::Types button = convertButtonCode(event.xbutton.
325  button);
326  buttons.remove(button);
327 
328  break;
329  }
330  case MotionNotify:
331  pointer_position.x = event.xmotion.x;
332  pointer_position.y = event.xmotion.y;
333 
334  break;
335  default:
336  break;
337  }
338  }
339 
340  glXSwapBuffers(display, window);
341  #elif defined(OS_WINDOWS)
342  MSG message;
343 
344  while (true) {
345  int result = PeekMessage(&message, NULL, 0, 0, PM_REMOVE);
346  if (result == 0) {
347  break;
348  }
349 
350  TranslateMessage(&message);
351  DispatchMessage(&message);
352  }
353 
354  SwapBuffers(window_context);
355  #endif
356 }

369 LRESULT CALLBACK OpenGlWindow::windowProcedure(HWND window, UINT message, WPARAM
370  word_parameter, LPARAM long_parameter)
371 {
372  switch (message) {
373  case WM_SIZE: {
374  size_t width = LOWORD(long_parameter);
375  size_t height = HIWORD(long_parameter);
376  if (width != size.x || height != size.y) {
377  resize(width, height);
378  }
379 
380  return 0;
381  }
382  #ifndef RAW_INPUT
383  case WM_KEYDOWN: {
384  KeyCode::Types key = convertKeyCode(word_parameter);
385  keys.push_back(key);
386 
387  return 0;
388  }
389  case WM_KEYUP: {
390  KeyCode::Types key = convertKeyCode(word_parameter);
391  keys.remove(key);
392 
393  return 0;
394  }
395  case WM_LBUTTONDOWN:
396  buttons.push_back(ButtonCode::BUTTON_LEFT);
397  return 0;
398  case WM_LBUTTONUP:
399  buttons.remove(ButtonCode::BUTTON_LEFT);
400  return 0;
401  case WM_MBUTTONDOWN:
402  buttons.push_back(ButtonCode::BUTTON_MIDDLE);
403  return 0;
404  case WM_MBUTTONUP:
405  buttons.remove(ButtonCode::BUTTON_MIDDLE);
406  return 0;
407  case WM_RBUTTONDOWN:
408  buttons.push_back(ButtonCode::BUTTON_RIGHT);
409  return 0;
410  case WM_RBUTTONUP:
411  buttons.remove(ButtonCode::BUTTON_RIGHT);
412  return 0;
413  case WM_MOUSEMOVE: {
414  POINTS pointer_position = MAKEPOINTS(long_parameter);
415  this->pointer_position.x = pointer_position.x;
416  this->pointer_position.y = pointer_position.y;
417 
418  return 0;
419  }
420  #else
421  case WM_INPUT: {
422  unsigned int required_size = 0;
423  GetRawInputData(reinterpret_cast<HRAWINPUT>(long_parameter),
424  RID_INPUT, NULL, &required_size, sizeof(RAWINPUTHEADER));
425 
426  unsigned char* data = new unsigned char[required_size];
427  unsigned int gotten_size = GetRawInputData(reinterpret_cast<
428  HRAWINPUT>(long_parameter), RID_INPUT, data, &required_size,
429  sizeof(RAWINPUTHEADER));
430  if (gotten_size != required_size) {
431  Console::error() << "AnnaGraphics warning: function "
432  "GetRawInputData() does not return correct size.";
433  }
434 
435  RAWINPUT* raw_data = reinterpret_cast<RAWINPUT*>(data);
436  if (raw_data->header.dwType == RIM_TYPEKEYBOARD) {
437  KeyCode::Types key = convertKeyCode(raw_data->data.keyboard.
438  VKey);
439  if (raw_data->data.keyboard.Flags & RI_KEY_BREAK) {
440  keys.remove(key);
441  } else {
442  keys.push_back(key);
443  }
444  } else if (raw_data->header.dwType == RIM_TYPEMOUSE) {
445  if (raw_data->data.mouse.usButtonFlags & \
446  RI_MOUSE_LEFT_BUTTON_DOWN)
447  {
448  buttons.push_back(ButtonCode::BUTTON_LEFT);
449  } else if (raw_data->data.mouse.usButtonFlags & \
450  RI_MOUSE_LEFT_BUTTON_UP)
451  {
452  buttons.remove(ButtonCode::BUTTON_LEFT);
453  } else if (raw_data->data.mouse.usButtonFlags & \
454  RI_MOUSE_MIDDLE_BUTTON_DOWN)
455  {
456  buttons.push_back(ButtonCode::BUTTON_MIDDLE);
457  } else if (raw_data->data.mouse.usButtonFlags & \
458  RI_MOUSE_MIDDLE_BUTTON_UP)
459  {
460  buttons.remove(ButtonCode::BUTTON_MIDDLE);
461  } else if (raw_data->data.mouse.usButtonFlags & \
462  RI_MOUSE_RIGHT_BUTTON_DOWN)
463  {
464  buttons.push_back(ButtonCode::BUTTON_RIGHT);
465  } else if (raw_data->data.mouse.usButtonFlags & \
466  RI_MOUSE_RIGHT_BUTTON_UP)
467  {
468  buttons.remove(ButtonCode::BUTTON_RIGHT);
469  }
470 
471  if (raw_data->data.mouse.usFlags & MOUSE_MOVE_ABSOLUTE) {
472  this->pointer_position.x = raw_data->data.mouse.lLastX;
473  this->pointer_position.y = raw_data->data.mouse.lLastY;
474  } else {
475  this->pointer_position.x += raw_data->data.mouse.lLastX;
476  this->pointer_position.y += raw_data->data.mouse.lLastY;
477  }
478  }
479 
480  delete[] data;
481  data = NULL;
482 
483  return 0;
484  }
485  #endif
486  default:
487  return DefWindowProc(window, message, word_parameter,
488  long_parameter);
489  }
490 }
